const cssId = 'rt-css-servicerow';
const css = `
.rt-svc {
  position: relative;
  display: grid;
  grid-template-columns: 70px minmax(0, 1.2fr) minmax(180px, 360px) 50px;
  align-items: center;

  border-top: 1px solid var(--border-default);

  cursor: pointer;

  isolation: isolate;
}
.rt-svc:last-of-type { border-bottom: 1px solid var(--border-default); }
.rt-svc::before {
  content: '';
  position: absolute;
  inset: 0;
  background: var(--accent);
  transform: scaleY(0);
  transform-origin: top;
  transition: transform 0.35s var(--ease-inout, ease);
  z-index: -1;
}
.rt-svc:hover::before { transform: scaleY(1); transform-origin: bottom; }
.rt-svc > * { min-width: 0; }
.rt-svc-num {
  font-family: var(--font-mono);
  font-size: 11px;

function inject() {
  if (typeof document !== 'undefined' && !document.getElementById(cssId)) {
    const el = document.createElement('style');
    el.id = cssId;
    el.textContent = css;
    document.head.appendChild(el);
  }
}

function ServiceRow({ num, title, desc, href = '#', onClick, style }) {
  inject();
  return (
    <a className="rt-svc" href={href} onClick={onClick} style={style}>
      <span className="rt-svc-num">/ {num}</span>
      <span className="rt-svc-title">{title}</span>
      <span className="rt-svc-desc">{desc}</span>
      <span className="rt-svc-arr">→</span>
    </a>
  );
}
